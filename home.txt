"use client";

import { useState } from "react";
import { motion } from "framer-motion";
import { Input } from "@/components/ui/input";
import { Button } from "@/components/ui/button";
import { Card, CardContent } from "@/components/ui/card";
import { Label } from "@/components/ui/label";
import { CheckCircle2, Sparkles, Shield, BellRing, ChevronRight } from "lucide-react";

export default function LandingPage() {
  const [loading, setLoading] = useState(false);
  const [success, setSuccess] = useState<null | "ok" | "error" >(null);

  async function onSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault();
    const form = e.currentTarget;
    const formData = new FormData(form);
    const payload = Object.fromEntries(formData.entries());

    try {
      setLoading(true);
      setSuccess(null);
      const res = await fetch("/api/whitelist", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload),
      });
      if (!res.ok) throw new Error("bad status");
      setSuccess("ok");
      form.reset();
    } catch (err) {
      setSuccess("error");
    } finally {
      setLoading(false);
    }
  }

  return (
    <main dir="rtl" className="min-h-screen bg-[#0B0B10] text-white antialiased relative overflow-hidden">
      {/* Subtle light layer */}
      <div className="absolute inset-0 -z-10 bg-[#0B0B10]" />

      {/* Header */}
      <header className="container mx-auto px-4 py-6 flex items-center justify-between">
        <div className="flex items-center gap-3">
          <div className="h-9 w-9 rounded-xl bg-white/10 grid place-items-center shadow-lg">
            <Sparkles className="h-5 w-5" />
          </div>
          <span className="text-lg font-semibold tracking-tight">مشروع سعودي — نسخة ما قبل الإطلاق</span>
        </div>
        <Button asChild variant="secondary" className="rounded-xl bg-white/10 hover:bg-white/20 border border-white/10 backdrop-blur">
          <a href="#join" className="flex items-center gap-1">
            انضم مبكرًا <ChevronRight className="h-4 w-4" />
          </a>
        </Button>
      </header>

      {/* HERO */}
      <section className="container mx-auto px-4 pt-6 pb-8">
        <div className="grid lg:grid-cols-2 gap-10 items-center">
          <div className="order-2 lg:order-1">
            <motion.h1
              initial={{ opacity: 0, y: 20 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ duration: 0.6 }}
              className="text-3xl md:text-5xl font-black leading-[1.1]"
            >
              قريبًا…
              <span className="block mt-2 text-white/90">
                حضورك الرقمي لن يكون كما كان من قبل.
              </span>
            </motion.h1>

            <motion.p
              initial={{ opacity: 0, y: 20 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: 0.1, duration: 0.6 }}
              className="mt-5 text-base md:text-lg text-white/70 max-w-xl"
            >
              فريق تقني وإبداعي يعمل بصمت على تجربة مختلفة كليًا. التفاصيل محجوبة الآن —
              لكن من ينضم مبكرًا سيعرف أولًا، وسيحصل على ميزة لا تتكرر.
            </motion.p>

            <ul className="mt-8 space-y-3 text-white/80">
              <li className="flex items-center gap-3">
                <CheckCircle2 className="h-5 w-5 text-emerald-400" />
                <span>ليست أداة تقليدية… بل نقلة في طريقة الحضور على الإنترنت.</span>
              </li>
              <li className="flex items-center gap-3">
                <CheckCircle2 className="h-5 w-5 text-emerald-400" />
                <span>أذكى، أقوى، وأبسط — لأصحاب المواقع من كل الأنواع.</span>
              </li>
              <li className="flex items-center gap-3">
                <CheckCircle2 className="h-5 w-5 text-emerald-400" />
                <span>وصول مبكر + مزايا خاصة للدفعة الأولى.</span>
              </li>
            </ul>
          </div>

          {/* Visual / Teaser Card */}
          <div className="order-1 lg:order-2">
            <motion.div
              initial={{ opacity: 0, scale: 0.98 }}
              animate={{ opacity: 1, scale: 1 }}
              transition={{ duration: 0.6 }}
            >
              <Card className="rounded-3xl border-white/10 bg-white/5 backdrop-blur-xl shadow-2xl">
                <CardContent className="p-6 md:p-8">
                  <div className="relative overflow-hidden rounded-2xl border border-white/10 bg-[#0B0B10] p-6">
                    <div className="flex items-center gap-3 mb-4">
                      <div className="h-9 w-9 rounded-xl bg-white/10 grid place-items-center border border-white/20">
                        <BellRing className="h-5 w-5 text-white/70" />
                      </div>
                      <p className="text-sm text-white/70">معاينة أولية</p>
                    </div>

                    <h3 className="text-2xl md:text-3xl font-bold leading-tight">
                      شيء كبير يُصنع بهدوء.
                    </h3>
                    <p className="mt-3 text-white/70">
                      لا نكشف التفاصيل اليوم. لكننا نَعِدُك أن حضورك القادم سيكون مختلفًا.
                    </p>

                    <div className="mt-6 grid grid-cols-1 sm:grid-cols-3 gap-3">
                      <div className="rounded-xl border border-white/10 p-4">
                        <p className="text-xs text-white/60">موثوقية</p>
                        <p className="font-semibold mt-1">بنية إنتاجية</p>
                      </div>
                      <div className="rounded-xl border border-white/10 p-4">
                        <p className="text-xs text-white/60">خصوصية</p>
                        <p className="font-semibold mt-1">تفاصيل محجوبة</p>
                      </div>
                      <div className="rounded-xl border border-white/10 p-4">
                        <p className="text-xs text-white/60">أفضلية</p>
                        <p className="font-semibold mt-1">دعوات محدودة</p>
                      </div>
                    </div>

                    <div className="mt-6 flex items-center gap-3 text-sm text-white/60">
                      <Shield className="h-4 w-4" />
                      <span>لن نشارك بريدك. الدعوات تُرسل على دفعات.</span>
                    </div>
                  </div>
                </CardContent>
              </Card>
            </motion.div>
          </div>
        </div>
      </section>

      {/* COLUMN LAYOUT SECTION */}
      <section className="container mx-auto px-4 pb-16">
        <div className="max-w-4xl mx-auto text-center mb-10">
          <h2 className="text-2xl md:text-3xl font-extrabold">لماذا الانضمام مبكرًا؟</h2>
          <p className="mt-3 text-white/70">نقدّم لمحة مقتضبة بلا تفاصيل تكشف المسار. هذا كل ما تحتاج معرفته الآن.</p>
        </div>

        {/* Three column layout */}
        <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
          <div className="rounded-2xl border border-white/10 p-6 bg-white/5">
            <h3 className="text-lg font-semibold">أولوية الوصول</h3>
            <p className="mt-2 text-white/70">دعوات مبكرة لتجربة مغلقة، ومزايا خاصة للدفعة الأولى.</p>
          </div>
          <div className="rounded-2xl border border-white/10 p-6 bg-white/5">
            <h3 className="text-lg font-semibold">ميزة تنافسية</h3>
            <p className="mt-2 text-white/70">حضور رقمي مختلف يمنحك أفضلية قبل كشف التفاصيل للعامة.</p>
          </div>
          <div className="rounded-2xl border border-white/10 p-6 bg-white/5">
            <h3 className="text-lg font-semibold">تصميم مبني على الثقة</h3>
            <p className="mt-2 text-white/70">نهج بسيط وهادئ، بدون ضوضاء تسويقية، واهتمام حقيقي بالخصوصية.</p>
          </div>
        </div>
      </section>

      {/* JOIN FORM */}
      <section id="join" className="container mx-auto px-4 pb-20">
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          whileInView={{ opacity: 1, y: 0 }}
          viewport={{ once: true }}
          transition={{ duration: 0.5 }}
          className="max-w-2xl mx-auto"
        >
          <Card className="rounded-3xl border-white/10 bg-white/5 backdrop-blur-xl">
            <CardContent className="p-6 md:p-8">
              <h2 className="text-2xl md:text-3xl font-extrabold">انضم إلى القائمة البيضاء</h2>
              <p className="mt-2 text-white/70">
                ضع بريدك الإلكتروني لتكون ضمن الدفعة الأولى من التجربة المغلقة.
              </p>

              <form onSubmit={onSubmit} className="mt-6 grid gap-5">
                <div className="grid gap-2">
                  <Label htmlFor="name">الاسم (اختياري)</Label>
                  <Input id="name" name="name" placeholder="مثال: خالد العلي" className="h-11 rounded-xl bg-white/5 border-white/15 focus-visible:ring-white/40" />
                </div>

                <div className="grid gap-2">
                  <Label htmlFor="email">البريد الإلكتروني</Label>
                  <Input id="email" name="email" type="email" required placeholder="you@example.com" className="h-11 rounded-xl bg-white/5 border-white/15 focus-visible:ring-white/40" />
                </div>

                <div className="grid gap-2">
                  <Label htmlFor="siteType">نوع الموقع أو النشاط (اختياري)</Label>
                  <select id="siteType" name="siteType" className="h-11 rounded-xl bg-white/5 border border-white/15 px-3 focus:outline-none focus:ring-2 focus:ring-white/40">
                    <option value="" className="bg-[#0B0B10]">— اختر —</option>
                    <option value="store" className="bg-[#0B0B10]">متجر إلكتروني</option>
                    <option value="company" className="bg-[#0B0B10]">شركة / مؤسسة</option>
                    <option value="blog" className="bg-[#0B0B10]">مدونة / محتوى</option>
                    <option value="personal" className="bg-[#0B0B10]">شخصي / بورتفوليو</option>
                    <option value="other" className="bg-[#0B0B10]">أخرى</option>
                  </select>
                </div>

                <div className="flex items-center gap-3">
                  <Button disabled={loading} type="submit" className="h-11 rounded-xl px-6 bg-white/20 hover:bg-white/30 text-white font-semibold shadow-lg">
                    {loading ? "جارٍ الإرسال…" : "أضفني إلى القائمة البيضاء"}
                  </Button>
                  {success === "ok" && (
                    <span className="text-emerald-400 text-sm">تم استلام طلبك. سنرسل الدعوات على دفعات.</span>
                  )}
                  {success === "error" && (
                    <span className="text-rose-400 text-sm">تعذّر الإرسال مؤقتًا. حاول لاحقًا.</span>
                  )}
                </div>
              </form>
            </CardContent>
          </Card>
        </motion.div>
      </section>

      {/* FOOTER */}
      <footer className="border-t border-white/10 bg-black/20">
        <div className="container mx-auto px-4 py-8 text-sm text-white/60 flex flex-col md:flex-row gap-3 items-center justify-between">
          <p>© 2025 — فكرة سعودية تُصنع بهدوء. جميع الحقوق محفوظة.</p>
          <div className="flex items-center gap-4">
            <a href="#join" className="hover:text-white/90 underline underline-offset-4">القائمة البيضاء</a>
            <span className="select-none">•</span>
            <a href="mailto:info@example.com" className="hover:text-white/90">تواصل</a>
          </div>
        </div>
      </footer>
    </main>
  );
}
